import pyorient	
import logging, traceback
import os.path
import time
from utils import connect_to_database, transform_to_orientdb_input

# init the logging system
current_time = time.strftime("%Y-%m-%d %H-%M-%S",time.localtime(time.time()));
logging.basicConfig(
	filename = os.path.join(os.getcwd(), 'log', current_time+'_log.txt'), 
	level = logging.DEBUG, 
	filemode = 'w', 
	format = '%(asctime)s - %(levelname)s: %(message)s'
	)

def translator(db, ws, file_path):

	file_name = os.path.split(file_path)[1]
	malware_name = file_name.split('$')[0]
	function = file_name.split('$')[1]

	construct_graph(ws, db, malware_name, function)


def construct_graph(ws, db, malware_name, function):	

	# create vertexs
	malware_vertex, new_vertexs = _construct_vertex(ws, db, malware_name, function)

	# create edges 
	# apis = db.query("select * from Api limit -1") # limit -1 means get all items
	_construct_edge(db, malware_vertex, function, new_vertexs)

									
# init the database
def init_database(db):
	db.command("ALTER DATABASE CUSTOM useLightweightEdges=false")

	# create class for Vertex and Edge
	db.command("create class Api extends V")
	db.command("create property Api.date DATETIME")	
	db.command("create property Api.id long")
	db.command("create class Argument_sharing extends E")
	db.command("create class Data_dependence extends E")

	db.command("create class Argument extends V")
	
	db.command("create class Rat extends V")	
	db.command("create class Belong_to extends E")	


# create vertexs with key-value pairs
def _construct_vertex(ws, db, malware_name, function):	

	## create Api vertexs
	new_vertexs = [] # apis vertexs just created 
	distinct_vertexs = [] # used to find duplicated vertexs
	logging.debug("#create Api vertexs:")

	for row in ws.rows[1:]:							
		key_value_pairs = _construct_key_value_pairs(distinct_vertexs, row)

		if key_value_pairs != "duplicated":
			create_node_command = "create Vertex Api set " + key_value_pairs						
			new_vertexs.append(db.command(create_node_command)[0])
			logging.debug(create_node_command)


	## create Rat vertexs
	malware_name = transform_to_orientdb_input(malware_name)	
	function = transform_to_orientdb_input(function)
	create_rat_vertex_command = "create Vertex Rat set name = {}".format(malware_name)
	malware_vertex = db.command(create_rat_vertex_command)[0]
	logging.debug("#create Rat vertexs:")
	logging.debug(create_rat_vertex_command)

	return malware_vertex, new_vertexs

# construct edge from nodes
def _construct_edge(db, malware_vertex, function, apis):

	## create Belong_to Edge	
	logging.debug("#create Belong_to edge:")
	for api in apis:
		create_edge_command = "create edge Belong_to from {} to {} set function={}".format(api._rid, malware_vertex._rid, function)
		db.command(create_edge_command)		
		logging.debug(create_edge_command)

    ## create Argument_sharding and Data_dependence edges
	## find relationships between apis, there are three types:
	## argument_sharing between two apis ---->  all args in api_a vs all args in api_b ; 
	## data dependence relationships from a to b ----> all args in api_a vs all retrun value in api_b;
	## data dependence relationships from b to a ----> all return value in api_a vs all args in api_b

	logging.debug("#create Argument_sharding and Data_dependence edges:")
	distinct_edges = []
	for api_a in apis:

		for arg_a in api_a.arguments:
			logging.debug("arg_a " + arg_a)
			if arg_a != "": # ignore empty string arg

				start_index = apis.index(api_a) + 1 # avoid repeat comparision, api_b starts from item next to api_a
				for api_b in apis[start_index:start_index+10]: # only search 1000 items				

					if api_a.api != api_b.api:      # ignore relatioship between two same API

						for arg_b in api_b.arguments: # find argument_sharing relationships
							if arg_a == arg_b: 
								# print "{}:{}({}) == {}:{}({}) \n".format(api_a.id, api_a.api, arg_a, api_b.id, api_b.api, arg_b)																
								new_edge = (api_a._rid, api_b._rid, transform_to_orientdb_input(arg_a.decode('utf-8')))
								if new_edge not in distinct_edges:
									create_edge_command = u"create Edge Argument_sharing from {} to {} set arg = {}".format(*new_edge)							
									db.command(create_edge_command)										
									logging.debug(create_edge_command)

									distinct_edges.append(new_edge)

			
						# if arg_a == api_b.return_value and api_a.date >= api_b.date and api_a.id >= api_b.id \
						# 		and api_b.return_value not in ['0x0000','0x00000000','0x0000000000000000']: # find data dependence relationships from a to b
						# 	# print "{}:{}({}) --> {}:{} \n".format(api_a.id, api_a.api, arg_a, api_b.id, api_b.return_value)														
						# 	new_edge = (api_a._rid, api_b._rid, transform_to_orientdb_input(arg_a.decode('utf-8')))
						# 	if new_edge not in distinct_edges:
						# 		create_edge_command = u"create Edge Data_dependence from {} to {} set arg = {}".format(*new_edge)
						# 		db.command(create_edge_command)														
						# 		logging.debug(create_edge_command)

						# 		distinct_edges.append(new_edge)
		
		start_index = apis.index(api_a) + 1		
		for api_b in apis[start_index:start_index+10] : # find data dependence relationships from b to a

			if api_a != api_b: 

				for arg_b in api_b.arguments:				

					if arg_b == api_a.return_value and api_b.date >= api_a.date and api_b.id >= api_a.id \
							and api_a.return_value not in ['0x0000','0x00000000','0x0000000000000000']: 
						# print "{}:{}({}) --> {}:{} \n".format(api_b.id, api_b.api, arg_b, api_a.id, api_a.return_value)
						new_edge = (api_b._rid, api_a._rid, transform_to_orientdb_input(arg_b.decode('utf-8')))
						if new_edge not in distinct_edges:
							create_edge_command = u"create Edge Data_dependence from {} to {} set arg = {}".format(*new_edge)
							db.command(create_edge_command)	
							logging.debug(create_edge_command)

							distinct_edges.append(new_edge)


# automatically constrcut key-value pairs for Vertex, such as id='1', api='getKeyState'..
def _construct_key_value_pairs(distinct_vertexs, row):

	keys = ["id", "date", "time_of_day", "relative_time", "thread", "tid", "module", "category", "api",
	"return_type", "return_value", "return_address", "error", "duration", "full_category"]

	key_value_pairs = u""

	logging.debug("    construct_key_value_pairs:")
	for key in keys:	
		value =  row[keys.index(key)].value		
		logging.debug("    {} : {} ".format(key, (u"{}".format(value)).encode('utf-8')))

		# split args from api calls`
		if key == "api": 		
			if value in distinct_vertexs:
				return "duplicated"
			else:
				distinct_vertexs.append(value)

			index_left_bracket = value.index('(')
			index_right_bracket = value.index(')')

			# api name
			api_name = value[:index_left_bracket] 
			key_value_pairs += u"{}={},".format(key,transform_to_orientdb_input(api_name))				

			# arguments
			arguments = value[index_left_bracket+1:index_right_bracket].split(",")    # extract the arguments from api call		
			arguments = [transform_to_orientdb_input(arg) for arg in arguments]       # transform the form of args into what orientdb accept
			key_value_pairs += u"{}=[{}],".format('arguments', u','.join(arguments))  # construct the command in key-value form

		else:					
			key_value_pairs += u"{}={},".format(key,transform_to_orientdb_input(value))	
			#print type(value),repr(value),"{}='{}',".format(key,value)	
		
	return key_value_pairs.strip(',') # remove the last ','


# automatically get all excel files(.xls or .xlsx) from a certain directory 
def get_trace_files(dir_path):
	
	trace_files = []

	import os
	files = os.listdir(dir_path)

	logging.info("Listing trace files:")
	for file in files:
		if file.endswith(".xlsx") or file.endswith('.xls'):
			trace_files.append(os.path.join(dir_path,file))
			logging.info(file)

	return trace_files

if __name__ == "__main__":		

	db = connect_to_database()
	try:		
		init_database(db)
		
		dir_path = os.path.join("..","excel_data")
		trace_files = get_trace_files(dir_path)

		for file_path in trace_files:
			from openpyxl import load_workbook
			wb = load_workbook(file_path)
			ws = wb.active
			translator(db, ws, file_path)
	except:	
		logging.error(traceback.format_exc())
	finally:
		db.db_close()

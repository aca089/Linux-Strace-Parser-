import pyorient	

def connect_to_database():

	# connect to database server
	db = pyorient.OrientDB("localhost", 2424)  
	db.connect("root", "list")

	# create the database. if the database exists, delete it and recreate one
	db_name = "traces"
	db_username = "admin"
	db_password = "admin"

	if db.db_exists( db_name, pyorient.STORAGE_TYPE_PLOCAL):
	    db.db_drop(db_name)	    

	db.db_create(db_name, pyorient.DB_TYPE_GRAPH, pyorient.STORAGE_TYPE_PLOCAL)	    
	    
	# connect to the database
	db.db_open( db_name, db_username, db_password )

	return db


# polish the data and make it acceptable to database
def transform_to_orientdb_input(data):
	# print repr(data)
	if isinstance(data,basestring):
		data = data.strip(" ") # strip space 	
		data = data.strip("\"") # strip double quotes
		data = data.strip("\'") # strip single quotes	

	data = u"{}".format(data) # convert data into string	

	data = data.replace("\\u","\\U") # prevent orientdb from escapingsome path as unicode string .i.e orienddb can't parse strring like set a=["C:\user\"] except a=["C:\\User"]

	import json
	data = json.dumps(data).strip("\"") 

	data = data.encode("unicode_escape") # encode data into unicode escape code which orientdb supports .i.e "C:\Users" -> "C:\\Users" 
	data = data.replace("\'", "\\\'") # escape the single quotes .i.e "'" -> "\'"
	data = data.replace("\"", "\\\"") # escape the double quotes .i.e '"' -> '\"'	

	data = u"'{}'".format(data) # add single quotes around data

	return data
